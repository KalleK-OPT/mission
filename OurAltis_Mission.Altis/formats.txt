Base format: [<ID>, <Side>, <Position>, <IsCamp>] - [STRING, SIDE, POSITION2D, BOOLEAN]
BaseList format: [Base, Base, ...]

InfantryType format: [<ClassName>, <Amount>] - [STRING, NUMBER]
InfantryCollection format: [InfantryType, InfantryType, ...]
BaseInfantryList format: [[<BaseID>, InfantryCollection], ...] - [[STRING, <>], ...]
SideInfantry format: [<Side>, BaseInfantryList] - [SIDE, <>]

Pos format: [<ID>, <Name>, <Position>] - [NUMBER, STRING, POSITION]
RespawnPositions format: [Pos, ...] - [<>]

Role format: [<ID>, <Availability>, <name>, <Create>] - [NUMBER, ARRAY of Strings, STRING, CODE] -> Create will be called in order to create the new unit (has to return it)
RespawnRoles format: [Role, ...] - [<>]
